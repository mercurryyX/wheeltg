<!-- Telegram Web App SDK -->

<script src="https://telegram.org/js/telegram-web-app.js"></script>

<script>

const tg = window.Telegram.WebApp;

tg.expand();


const segments = ["10₽", "20₽", "50₽", "100₽", "Попробуй снова", "200₽"];

const wheel = document.getElementById("wheel");

const result = document.getElementById("result");


// создаём сегменты колеса

segments.forEach((label, i) => {

  const seg = document.createElement("div");

  seg.className = "segment";

  const angle = (360 / segments.length) * i;

  seg.style.transform = `rotate(${angle}deg)`;

  seg.style.background = `hsl(${angle}, 80%, 50%)`;

  seg.textContent = label;

  wheel.appendChild(seg);

});


document.getElementById("spinBtn").addEventListener("click", () => {

  const spins = Math.floor(Math.random() * 5) + 5; // 5-9 оборотов

  const segmentAngle = 360 / segments.length;

  const randIndex = Math.floor(Math.random() * segments.length);

  const rotateDeg = spins * 360 + randIndex * segmentAngle + segmentAngle / 2;


  wheel.style.transition = "transform 5s cubic-bezier(0.33, 1, 0.68, 1)";

  wheel.style.transform = `rotate(${rotateDeg}deg)`;


  // показываем результат через 5 секунд

  setTimeout(() => {

    result.textContent = `Вы выиграли: ${segments[randIndex]}!`;


    // отправляем результат боту (если нужно)

    tg.sendData(JSON.stringify({ prize: segments[randIndex] }));

  }, 5000);

});

</script>
